用JS写一个数组去重的方法 

今天看到一道面试题是说用 Javascript 写一个数组去重的方法，我也做了一下，思路大概是先排序，再用另一个数组来存不重复项。代码如下：

function removeDuplicates1(array1) {
        var array = [];
        var b = 0;
        var array2 = array1.sort();
        for (var i = 0; i < array2.length; i++) {
            if (array2[i] === array2[i+1]) continue;
            else array[b] = array2[i];
            b++;
        }
        return array;
    }

方法二：利用数组下标和临时变量


var removeDuplicates2 = function(array){
    var array2 = array.sort();
    var temp = array2[0];
    var a = 1;
    for (var i = 1; i < array2.length; i++) {
        if(temp === array2[i]) continue;
        else{
            temp = array2[i];
            array2[a] = temp;
            a++;
        }
    }
    return array2.slice(0,a);
}
