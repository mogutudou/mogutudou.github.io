穷举法简析Google方程式

google方程式的题目是这样的：有一个字符组成的等式：WWWDOT - GOOGLE = DOTCOM，每个字符代表一个0－9之间的数字，WWWDOT、GOOGLE和DOTCOM都是合法的数字，不能以0开头。请找出一组字符和数字的对应关系，使它们互相替换，并且替换后的数字能够满足等式。将方程式改成竖式可能更直观一些：

下面我们就用 JavaScript 来找到符合条件的等式，然后用alert将等式显示在网页中。（本人比较笨，目前只写出思维易懂非递归的穷举法，有其他更好方法的读者还请联系我，抱着一颗乐于学习的心态我们一起探讨）

function google(){
        for(var w=1;w<10;w++){
        for(var d=1;d<10;d++){
            if(d==w) continue;
            for(var o=0;o<10;o++){
                if(o==w||o==d) continue;
                for(var t=0;t<10;t++){
                    if(t==w||t==d||t==o) continue;
                    for(var g=1;g<10;g++){
                        if(g==w||g==d||g==o||g==t) continue;
                        for(var l=0;l<10;l++){
                            if(l==w||l==d||l==o||l==t||l==g) continue;
                            for(var e=0;e<10;e++){
                                if(e==w||e==d||e==o||e==t||e==g||e==l) continue;
                                for(var c=0;c<10;c++){
                                    if(c==w||c==d||c==o||c==t||c==g||c==l||c==e) continue;
                                    for(var m=0;m<10;m++){
                                        if(m==w||m==d||m==o||m==t||m==g||m==l||m==e||m==c) continue;
                                            var wwwdot = w100000 + w10000 + w1000 + d100 + o10 + t;
                                            var google = g100000 + o10000 + o1000 + g100 + l10 + e;
                                            var dotcom = d100000 + o10000 + t1000 + c100 + o*10 + m;
                                            if (wwwdot - google === dotcom) {
                                                alert(wwwdot + “-“ + google + “=” + dotcom);
                                        }
                                    }
                                }
                            }
                        }
                    }
                }
            }
        }
    }
    }
    window.onload = google;
