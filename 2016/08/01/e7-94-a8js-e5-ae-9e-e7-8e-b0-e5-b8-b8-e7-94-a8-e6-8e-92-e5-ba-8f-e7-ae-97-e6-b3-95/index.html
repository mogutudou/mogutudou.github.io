用JS实现常用排序算法 

看到排序算法这里，发现有些面试官喜欢考这个，所以我大概总结一下，也为自己做好准备，以下内容为我个人见解，如有错误的地方还欢迎指正。以下例子均是给一个数字数组按从小到大排序

一、js原生有sort()方法

function paixu(array){
    return array.sort(function(a,b){return (a-b)})
}

二、冒泡排序

function paixu2(array){
    var temp;
    for (var i = 0; i < array.length; i++) {
        for (var j = i+1; j < array.length; j++) {
                    if(array[i] > array[j]){
                temp = array[i];
                array[i] = array[j];
                array[j] = temp;
            }
        }
    }
    return array;
}

三、插入排序

function paixu3(array){
    for (var i = 0; i < array.length; i++) {
            var temp = array[i];
            var j = i - 1; 
            while (j >= 0 && array[j] > temp) {
                array[j + 1] = array[j];
                j--;
            }
            array[j + 1] = temp;
        }
        return array;
    }
四、快排 
function kuaipai(array){
    if (array.length <= 1) {return array;}
    var base = array.splice(0,1)[0];
    var left = [];
    var right = [];
    for (var i = 0; i < array.length; i++) {
        if(array[i] < base){
            left.push(array[i]);
        } else{
            right.push(array[i]);
        }
    }
    return kuaipai(left).concat([base],kuaipai(right));
}
